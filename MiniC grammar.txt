program ~ decl_list
decl_list ~ decl_list decl | decl
decl ~ var_decl | fun_decl
var_decl ~ type_spec ID SEMICOLON | type_spec ID LEFT_SQUARE_B RIGHT_SQUARE_B SEMICOLON
type_spec ~ VOID | BOOL | INT | FLOAT
fun_decl ~ type_spec ID LEFT_ROUND_B params RIGHT_ROUND_B compound_stmt
params ~ param_list | VOID
param_list ~ param_list COMMA param | param
param ~ type_spec ID | type_spec ID LEFT_SQUARE_B RIGHT_SQUARE_B
compound_stmt ~ LEFT_CURLY_B local_decls stmt_list RIGHT_CURLY_B
stmt_list ~ stmt_list stmt | EMPTY
stmt ~ expr_stmt | compound_stmt | if_stmt | while_stmt | return_stmt | break_stmt
expr_stmt ~ expr SEMICOLON | SEMICOLON
while_stmt ~ WHILE LEFT_ROUND_B expr RIGHT_ROUND_B stmt
local_decls ~ local_decls local_decl | EMPTY
local_decl ~ type_spec ID SEMICOLON | type_spec ID LEFT_SQUARE_B RIGHT_SQUARE_B SEMICOLON
if_stmt ~ IF LEFT_ROUND_B expr RIGHT_ROUND_B stmt | IF LEFT_ROUND_B expr RIGHT_ROUND_B stmt ELSE stmt
return_stmt ~ RETURN SEMICOLON | RETURN expr SEMICOLON
expr ~ ID ASSIGNMENT expr | ID LEFT_SQUARE_B expr RIGHT_SQUARE_B ASSIGNMENT expr | expr OR expr| expr EQUAL expr | expr NOT EQUAL expr| expr LESS THAN OR EQUAL expr | expr LESSTHAN expr | expr GREATER THAN OR EQUAL expr | expr GREATER THAN expr| expr AND expr| expr PLUS expr | expr MINUS expr| expr MULTIPLY expr | expr DIVIDE expr | expr MOD expr| NOT expr | - expr | + expr | LEFT_ROUND_B expr RIGHT_ROUND_B | ID | ID LEFT_SQUARE_B expr RIGHT_SQUARE_B | ID LEFT_ROUND_B args RIGHT_ROUND_B | ID DOT size| BOOL_LIT | INT_LIT | FLOAT_LIT | NEW type_spec LEFT_SQUARE_B expr RIGHT_SQUARE_B
args ~ arg_list | EMPTY
arg_list ~ arg_list COMMA expr | expr
